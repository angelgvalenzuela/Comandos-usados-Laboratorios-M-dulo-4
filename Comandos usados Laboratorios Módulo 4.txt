COMANDOS USADOS PRACTICA 1: INSTALAR UN SERVIDOR HTTP
sudo su
dnf install httpd
systemctl enable --now httpd
systemctl status httpd
# --- PARTE 1: SITIO HOLA MUNDO ---
mkdir -p /var/www/html/hola
cd /var/www/html/hola
nano index.html
# [CONTENIDO DE: index.html]
# ----------------------------------------------------
<!DOCTYPE html>
<html>
<head><tittle>HOLA MUNDO</tittle><head>
<body>
<h1>HOLA MUNDO<h1>
</body>
<!html>
# ----------------------------------------------------
nano /etc/httpd/conf.d/hola.conf
# [CONTENIDO DE: hola.conf]
# ----------------------------------------------------
  <VirtualHost *:80>
      ServerName hola.local
      DocumentRoot /var/www/html/hola
      <Directory /var/www/html/hola>
          AllowOverride All
          Require all granted
      </Directory>
 
      ErrorLog /var/log/httpd/hola_error.log
      CustomLog /var/log/httpd/hola_access.log combined
  </VirtualHost>
# ----------------------------------------------------
nano /etc/hosts
# (Se agrega la resolución local al final: 127.0.0.1 hola.local)
systemctl restart httpd
# (Comprobación en Firefox con la IP de la VM)
